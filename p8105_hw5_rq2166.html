fill_na = function(x) {
  
  if (is.numeric(x)) {
    replace = round((mean(x, na.rm = TRUE)),2)
    x[is.na(x)] <- replace
  } else if (is.character(x)) {
    replace = "virginica"
    x[is.na(x)] <- replace
  }
  x
}
